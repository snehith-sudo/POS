 package com.increff.project.dao;
 
 import com.increff.project.entity.InventoryPojo;
 import com.increff.project.entity.OrderPojo;
 import com.increff.project.model.constants.OrderStatus;
 import com.increff.project.entity.SalesReportPojo;
 import jakarta.persistence.EntityManager;
 import jakarta.persistence.PersistenceContext;
 
 import java.time.ZonedDateTime;
 import java.util.List;
 
 public abstract class AbstractDAO<T> {
 
     @PersistenceContext
     protected EntityManager em;
 
     private final Class<T> entityClass;
 
     protected AbstractDAO(Class<T> entityClass) {
         this.entityClass = entityClass;
     }
 
     protected void insert(T entity) {
         em.persist(entity);
     }
 
     protected List<T> findByClientId(Integer clientId,int page,int size) {
 
         return em.createQuery(
                         "SELECT e FROM " + entityClass.getSimpleName() +
                                 " e WHERE e.clientId = :clientId",
                         entityClass
                 )
                 .setParameter("clientId", clientId)
                 .setFirstResult(page * size)
                 .setMaxResults(size)
                 .getResultList();
     }
 
     protected long countItemsOfClientById(Integer clientId){
         return em.createQuery("SELECT COUNT(e) FROM "+entityClass.getSimpleName()+
                 " e WHERE e.clientId = :clientId",Long.class)
                 .setParameter("clientId",clientId)
                 .getSingleResult();
     }
 
     protected T findById(Integer id) {return em.find(entityClass, id);}
 
     protected T findByName(String name) {
 
         List<T> results = em.createQuery(
                         "SELECT e FROM " + entityClass.getSimpleName() + " e WHERE e.name = :name",
                         entityClass)
                 .setParameter("name", name)
                 .getResultList();
 
         return results.isEmpty() ? null : results.getFirst();
     }
 
     protected boolean existsById(Integer id) {return em.find(entityClass, id) != null;}
 
     protected List<String> selectmultipleExistingNames(List<String> ids) {
 
         if (ids == null || ids.isEmpty()) {return List.of();}
 
         String jpql = "SELECT e.name FROM " + entityClass.getSimpleName() + " e WHERE e.name IN :ids";
 
         return em.createQuery(jpql, String.class)
                 .setParameter("ids", ids)
                 .getResultList();
     }
 
     protected List<T> selectExistingEntityWithIds(List<Integer> ids) {
 
         if (ids == null || ids.isEmpty()) {return List.of();}
 
         String jpql = "SELECT e FROM " + entityClass.getSimpleName() + " e WHERE e.id IN :ids";
 
         return em.createQuery(jpql, entityClass)
                 .setParameter("ids", ids)
                 .getResultList();
     }
 
     protected T findByBarcode(String barcode, Object value) {
 
         String jpql = "SELECT e FROM " + entityClass.getSimpleName()
                 + " e WHERE e." + barcode + " = :value";
 
         List<T> result = em.createQuery(jpql, entityClass)
                 .setParameter("value", value)
                 .getResultList();
 
         return result.isEmpty() ? null : result.get(0);
     }
 
     protected List<T> getByStatus(OrderStatus status,int page,int size) {
 
         String jpql =
                 "SELECT e FROM " + entityClass.getSimpleName() + " e " +
                         "WHERE e.status = :status";
 
         return em.createQuery(jpql, entityClass)
                 .setParameter("status", status)
                 .setFirstResult(page * size)
                 .setMaxResults(size)
                 .getResultList();
     }
 
     protected long countOrderByStatus(OrderStatus status){
         String jpql = "SELECT COUNT(e) FROM "+ entityClass.getSimpleName()+" e "
                 +"WHERE e.status =:status";
 
         return em.createQuery(jpql,Long.class)
                 .setParameter("status",status)
                 .getSingleResult();
     }
 
     protected void update(T entity){
         em.merge(entity);
     }
 
     // MUST be protected
     protected List<T> selectAll() {
         return em.createQuery("SELECT e FROM " + entityClass.getSimpleName() + " e ORDER BY e.id", entityClass)
                 .getResultList();
     }
 
     protected List<T> findBetweenDates(ZonedDateTime startDate, ZonedDateTime endDate,int page,int size) {
 
         String jpql =
                 "SELECT e FROM " + entityClass.getSimpleName() + " e " +
                         "WHERE e.orderTime BETWEEN :startDate AND :endDate";
 
         return em.createQuery(jpql, entityClass)
                 .setParameter("startDate", startDate)
                 .setParameter("endDate", endDate)
                 .setFirstResult(page * size)
                 .setMaxResults(size)
                 .getResultList();
     }
 
     protected long countRowsBetweenDates(ZonedDateTime startDate, ZonedDateTime endDate){
         String jpql = "SELECT COUNT(e) FROM "+ entityClass.getSimpleName() + " e "+
                 "WHERE e.orderTime BETWEEN :startDate AND :endDate";
 
         return em.createQuery(jpql,Long.class)
                 .setParameter("startDate",startDate)
                 .setParameter("endDate",endDate)
                 .getSingleResult();
     }
 
     protected List<T> selectPaged(int offset, int limit, String orderBy) {
         String jpql = "SELECT e FROM " + entityClass.getSimpleName() + " e ORDER BY e." + orderBy;
 
         return em.createQuery(jpql, entityClass)
                 .setFirstResult(offset)
                 .setMaxResults(limit)
                 .getResultList();
     }
 
     protected long countAllRowsInTable() {
         String q = "SELECT COUNT(e) FROM " + entityClass.getSimpleName() + " e";
         return em.createQuery(q, Long.class).getSingleResult();
     }
 
     protected List<T> selectAllPaginated(int page, int size) {
 
         String orderBy;
 
         if (entityClass.equals(InventoryPojo.class)) {
             orderBy = "e.quantity ASC";
         } else if (entityClass.equals(OrderPojo.class) || (entityClass.equals(SalesReportPojo.class))) {
             orderBy = "e.id ASC";
         } else {
             orderBy = "e.name ASC";
         }
 
         String jpql = "SELECT e FROM " + entityClass.getSimpleName() +
                 " e ORDER BY " + orderBy;
 
         return em.createQuery(jpql, entityClass)
                 .setFirstResult(page * size)
                 .setMaxResults(size)
                 .getResultList();
     }
 
 }
