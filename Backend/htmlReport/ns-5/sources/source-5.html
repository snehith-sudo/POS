 package com.increff.project.dto;
 
 
 import java.util.List;
 import java.util.Locale;
 import java.util.Map;
 import java.util.stream.Collectors;
 
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.data.domain.Page;
 import org.springframework.data.domain.PageImpl;
 import org.springframework.stereotype.Component;
 
 import com.increff.project.entity.ClientPojo;
 import com.increff.project.entity.ProductPojo;
 import com.increff.project.model.data.ProductData;
 import com.increff.project.model.form.ProductFilterForm;
 import com.increff.project.model.form.ProductForm;
 import com.increff.project.model.form.TsvForm;
 import com.increff.project.useCase.ProductUseCase;
 import com.increff.project.utils.helpers.AbstractValidateNormalize;
 import com.increff.project.model.form.PageSizeForm;
 import static com.increff.project.utils.helpers.ProductHelper.convertFormToEntity;
 import com.increff.project.utils.utils.ProductUtil;
 import static com.increff.project.utils.utils.ProductUtil.convertProductEntityToData;
 
 @Component
 public class ProductDTO extends AbstractValidateNormalize {
 
     @Autowired
     private ProductFlow productFlow;
 
     /* ---------------- CREATE ---------------- */
     public void addProduct(ProductForm form)
     {
         validateNormalizeProducts(form);
         ProductPojo newproduct = convertFormToEntity(form);
         productFlow.addProduct(newproduct,(form.getClientName()).toLowerCase(Locale.ROOT));
     }
 
     public void updateProduct(ProductForm form){
         validateNormalizeProducts(form);
         ProductPojo updatedproduct = convertFormToEntity(form);
         productFlow.updateProduct(updatedproduct,form.getClientName());
     }
 
     public ProductData findByBarcode(ProductFilterForm form)
     {
         return convertProductEntityToData((productFlow.findByBarcode(form.getBarcode())));
     }
 
     public Page<ProductData> getAllClientProducts(ProductFilterForm form) {
         Page<ProductPojo> productPojoPage =
                 productFlow.getAllClientProducts(form.getClientName(),form.getPage(),form.getSize());
 
         List<ProductData> dataList = productPojoPage.getContent()
                 .stream()
                 .map(ProductUtil::convertProductEntityToData)
                 .collect(Collectors.toList());
 
         return new PageImpl<>(
                 dataList,
                 productPojoPage.getPageable(),
                 productPojoPage.getTotalElements()
         );
     }
 
     public Page<ProductData> getAllOrders(PageSizeHelper form) {
 
         Page<ProductPojo> productPojoPage = productFlow.getAllOrders(form.getPage(), form.getSize());
         List<ClientPojo> clients = productFlow.selectExistingClientIds(productPojoPage);
 
         Map<Integer, String>
                 clientIdToName = clients.stream()
                                  .collect(Collectors.toMap(ClientPojo::getId, ClientPojo::getName));
 
         List<ProductData> dataList = productPojoPage.getContent()
                 .stream()
                 .map(productPojo -> {
                     ProductData data = convertProductEntityToData(productPojo);
 
                     // set the clientName using the map
                     String clientName = clientIdToName.get(productPojo.getClientId());
                     data.setClientName(clientName);
 
                     return data;
                 })
                 .collect(Collectors.toList());
 
 
         return new PageImpl<>(
                 dataList,
                 productPojoPage.getPageable(),
                 productPojoPage.getTotalElements()
         );
     }
 
 
     public Map<String, String> addMultipleProduct(TsvForm form)
     {
        return productFlow.addMultipleProducts(form.getBase64file());
     }
 
     public void validateNormalizeProducts(ProductForm form)
     {
         validateNormalizeString(form.getBarcode(),"Barcode");
         validateNormalizeString(form.getName(),"Product Name");
         validateNormalizeString(form.getClientName(),"Client Name");
         validateNormalizeDouble(form.getMrp(),"Product Mrp");
     }
 
 }
